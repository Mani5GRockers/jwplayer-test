Set poster image for video element.

Source:

Audio only MP3 stream via HLS.

Expected behavior

Poster image should be visible during audio playback.
Playback should not fail due to cross domain issues while loading the poster image.

Settings:

jwplayer('player').setup({
    "file": "https://streaming4.ur.se/urse/_definst_/mp3:196000-196999/196066-4.mp3/playlist.m3u8",
    "image": "http://assets.ur.se/id/196066/images/1_l.jpg",
}).on('ready', function() {
    // Set poster image for video element to avoid black background for audio-only programs.
    var video = document.getElementById(this.id).getElementsByTagName('video')[0];
    if (video) video.setAttribute('poster', this.getConfig().image);
});
